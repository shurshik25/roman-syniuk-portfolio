# GitHub токен для автоматичного збереження змін
# Створіть .env файл в корені проекту з цим вмістом:

VITE_GITHUB_TOKEN=your_github_token_here
GITHUB_REPO=shurshik25/roman-syniuk-portfolio
GITHUB_BRANCH=main

# ІНСТРУКЦІЯ:
# 1. Перейменуйте цей файл в .env
# 2. Створіть GitHub Personal Access Token
# 3. Права: repo, workflow
# 4. НЕ комітьте .env файл в Git!
# 5. Додайте .env в .gitignore

# Як створити GitHub токен:
# 1. GitHub.com → Settings → Developer settings → Personal access tokens → Tokens (classic)
# 2. Generate new token → Select scopes: repo, workflow
# 3. Скопіюйте токен та вставте в .env файл
